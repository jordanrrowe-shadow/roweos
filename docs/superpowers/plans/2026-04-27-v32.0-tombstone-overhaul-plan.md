```javascript
// v32.0-A: Universal delete API. All deletes from any category go through these
// three functions. Per-category map of last local save times replaces the global
// var that the v27.0 grace period used.

var lastLocalSaveTime = {}; // { categoryId: timestampMs }
var LOCAL_SAVE_GRACE_PERIOD_DEFAULT = 5000;
window.lastLocalSaveTime = lastLocalSaveTime;

function _readArray(localStorageKey) {
  try {
    var raw = localStorage.getItem(localStorageKey);
    if (!raw) return [];
    var parsed = JSON.parse(raw);
    return Array.isArray(parsed) ? parsed : [];
  } catch (e) {
    return [];
  }
}

function _writeArray(localStorageKey, arr) {
  try {
    localStorage.setItem(localStorageKey, JSON.stringify(arr));
  } catch (e) {
    console.warn('[v32.0-A] _writeArray failed for ' + localStorageKey, e);
  }
}

function _readTombstoneSet(cat) {
  var arr = _readArray(cat.tombstoneKey);
  var set = {};
  for (var i = 0; i < arr.length; i++) set[arr[i]] = true;
  return set;
}

function _writeTombstoneSet(cat, set) {
  var arr = [];
  for (var k in set) if (set.hasOwnProperty(k)) arr.push(k);
  _writeArray(cat.tombstoneKey, arr);
}

function _runRerenders(cat) {
  if (!cat.rerender || !cat.rerender.length) return;
  for (var i = 0; i < cat.rerender.length; i++) {
    var fnName = cat.rerender[i];
    try {
      if (typeof window[fnName] === 'function') window[fnName]();
    } catch (e) {
      console.warn('[v32.0-A] rerender ' + fnName + ' threw', e);
    }
  }
}

// Resolve a function name like "writeDB" into the actual function. We look it up
// at call time (not at registry construction) so circular dependencies don't
// break us — 22-firebase-sync.js loads before 22a but runtime call order is
// what matters.
function _fn(name) {
  return (typeof window[name] === 'function') ? window[name] : null;
}

// Delete the cloud doc(s) for a category-id pair.
function _deleteCloudItem(cat, itemId) {
  return new Promise(function(resolve) {
    var deleteDBDoc = _fn('deleteDBDoc');
    var writeDB = _fn('writeDB');
    if (cat.cloudShape === 'subcollection') {
      if (!deleteDBDoc) { resolve({ ok: false, error: 'deleteDBDoc unavailable' }); return; }
      try {
        var p = deleteDBDoc(cat.cloudPath, itemId);
        if (p && typeof p.then === 'function') {
          p.then(function() { resolve({ ok: true }); }, function(err) { resolve({ ok: false, error: err }); });
        } else {
          resolve({ ok: true });
        }
      } catch (e) { resolve({ ok: false, error: e }); }
    } else if (cat.cloudShape === 'blob') {
      // Re-write the blob doc minus the item; readers expect blobField = array
      if (!writeDB) { resolve({ ok: false, error: 'writeDB unavailable' }); return; }
      try {
        var local = _readArray(cat.localKey);
        var payload = {};
        payload[cat.blobField] = local;
        payload._modifiedAt = Date.now();
        var p2 = writeDB(cat.cloudPath, payload);
        if (p2 && typeof p2.then === 'function') {
          p2.then(function() { resolve({ ok: true }); }, function(err) { resolve({ ok: false, error: err }); });
        } else {
          resolve({ ok: true });
        }
      } catch (e2) { resolve({ ok: false, error: e2 }); }
    } else if (cat.cloudShape === 'inline-field') {
      // Re-write the parent doc filtering its field array
      if (!writeDB) { resolve({ ok: false, error: 'writeDB unavailable' }); return; }
      try {
        var local2 = _readArray(cat.localKey);
        var pld = {};
        pld[cat.blobField] = local2;
        pld._modifiedAt = Date.now();
        var p3 = writeDB(cat.parentDoc, pld);
        if (p3 && typeof p3.then === 'function') {
          p3.then(function() { resolve({ ok: true }); }, function(err) { resolve({ ok: false, error: err }); });
        } else {
          resolve({ ok: true });
        }
      } catch (e3) { resolve({ ok: false, error: e3 }); }
    } else {
      resolve({ ok: false, error: 'unknown cloudShape: ' + cat.cloudShape });
    }
  });
}

function _writeCloudTombstone(cat, set) {
  var writeDB = _fn('writeDB');
  if (!writeDB) return Promise.resolve({ ok: false, error: 'writeDB unavailable' });
  var ids = [];
  for (var k in set) if (set.hasOwnProperty(k)) ids.push(k);
  return Promise.resolve(writeDB(cat.cloudTombstonePath, { ids: ids, _modifiedAt: Date.now() }))
    .then(function() { return { ok: true }; }, function(err) { return { ok: false, error: err }; });
}

function tombstoneAndDelete(categoryIdOrLabel, itemId) {
  var cat = getCategoryById(categoryIdOrLabel);
  if (!cat) return Promise.resolve({ ok: false, error: 'unknown category: ' + categoryIdOrLabel });
  if (itemId === undefined || itemId === null || itemId === '') {
    return Promise.resolve({ ok: false, error: 'itemId required' });
  }

  // 1. Append to local tombstone set
  var set = _readTombstoneSet(cat);
  set[itemId] = true;
  _writeTombstoneSet(cat, set);

  // 2. Filter local data array
  if (cat.localKey) {
    var localArr = _readArray(cat.localKey);
    var filtered = localArr.filter(function(item) {
      return item && item[cat.idField] !== itemId;
    });
    _writeArray(cat.localKey, filtered);
  }

  // 3. Update grace period
  lastLocalSaveTime[cat.id] = Date.now();

  // 4 & 5. Push tombstone + delete cloud doc(s) in parallel
  return Promise.all([
    _writeCloudTombstone(cat, set),
    _deleteCloudItem(cat, itemId)
  ]).then(function(results) {
    var tombRes = results[0];
    var delRes = results[1];
    var ok = tombRes.ok && delRes.ok;

    // 6. Trigger rerenders
    _runRerenders(cat);

    return {
      ok: ok,
      id: itemId,
      ts: Date.now(),
      tombstoneOk: tombRes.ok,
      deleteOk: delRes.ok,
      error: ok ? null : (tombRes.error || delRes.error)
    };
  });
}
window.tombstoneAndDelete = tombstoneAndDelete;

function tombstoneAndDeleteMany(categoryIdOrLabel, itemIds) {
  var cat = getCategoryById(categoryIdOrLabel);
  if (!cat) return Promise.resolve({ ok: false, count: 0, failed: [], error: 'unknown category' });
  if (!Array.isArray(itemIds) || !itemIds.length) {
    return Promise.resolve({ ok: true, count: 0, failed: [] });
  }

  // 1. Append all to local tombstone set
  var set = _readTombstoneSet(cat);
  for (var i = 0; i < itemIds.length; i++) set[itemIds[i]] = true;
  _writeTombstoneSet(cat, set);

  // 2. Filter local data array
  if (cat.localKey) {
    var localArr = _readArray(cat.localKey);
    var killSet = {};

// v32.0-A: Pull-side enforcement. Called by loadFromFirebaseV2 and every
// onSnapshot listener BEFORE the payload is merged into local state. Loads the
// union of local+cloud tombstones, writes the union back to converge devices,
// and filters the incoming payload (and any local dupes) against the union.

function applyTombstoneFilter(categoryIdOrLabel, cloudItems) {
  var cat = getCategoryById(categoryIdOrLabel);
  if (!cat) return cloudItems || [];
  if (!Array.isArray(cloudItems)) cloudItems = [];

  // 1. Read local tombstones
  var localSet = _readTombstoneSet(cat);

  // 2. Best-effort read cloud tombstones (we don't await it — readDB may be
  // async and we already have local state). The cloud merge is opportunistic;
  // the cloud-write side runs in tombstoneAndDelete already.
  var readDB = _fn('readDB');
  if (readDB) {
    try {
      var maybe = readDB(cat.cloudTombstonePath);
      // Some readDB implementations are sync (cache); some are async. Handle both.
      if (maybe && maybe.ids && Array.isArray(maybe.ids)) {
        for (var i = 0; i < maybe.ids.length; i++) localSet[maybe.ids[i]] = true;
      } else if (maybe && typeof maybe.then === 'function') {
        // Fire-and-forget: result will be picked up next pull
        maybe.then(function(res) {
          if (res && res.ids && Array.isArray(res.ids)) {
            var s = _readTombstoneSet(cat);
            var changed = false;
            for (var k = 0; k < res.ids.length; k++) {
              if (!s[res.ids[k]]) { s[res.ids[k]] = true; changed = true; }
            }
            if (changed) _writeTombstoneSet(cat, s);
          }
        }, function() {});
      }
    } catch (e) { /* ignore */ }
  }

  // 3. Persist union back to local
  _writeTombstoneSet(cat, localSet);

  // 4. Filter cloud items
  var filteredCloud = cloudItems.filter(function(item) {
    return item && !localSet[item[cat.idField]];
  });

  // 5. Filter local data array (catches dupes that slipped past)
  if (cat.localKey) {
    var localArr = _readArray(cat.localKey);
    var localFiltered = localArr.filter(function(item) {
      return item && !localSet[item[cat.idField]];
    });
    if (localFiltered.length !== localArr.length) {
      _writeArray(cat.localKey, localFiltered);
    }
  }

  return filteredCloud;
}

  for (var j = 0; j < keys.length; j++) {
    try {
      var v = localStorage.getItem(keys[j]);
      if (v !== null) snap[keys[j]] = v;
    } catch (e) { /* ignore */ }
  }
  snap._capturedAt = new Date().toISOString();
  snap._version = (window.ROWEOS_VERSION || 'unknown');
  return snap;
}

function writePrePurgeBackup_v32() {
  var snap = _snapshotForBackup();
  try { localStorage.setItem('roweos_pre_pull_backup', JSON.stringify(snap)); } catch (e) {}
  try { localStorage.setItem('roweos_pre_purge_backup_v32', JSON.stringify(snap)); } catch (e2) {}
  var writeDB = _fn('writeDB');
  if (!writeDB) return Promise.resolve({ ok: false, error: 'writeDB unavailable' });
  return Promise.resolve(writeDB('profile/preBackup_v32', { snapshot: snap, _modifiedAt: Date.now() }))

// Purge function. Runs ALL categories with legacyHeuristic. Does NOT prompt —
// the caller is responsible for confirmation. Use the modal in B.3 to wrap this.
function purgeLegacyFocusResidue(opts) {
  opts = opts || {};
  var force = opts.force === true;
  if (!force && localStorage.getItem('roweos_focus_purge_v32') === 'done') {
    return Promise.resolve({ ok: true, skipped: true, message: 'already purged' });
  }
  return writePrePurgeBackup_v32().then(function(backupRes) {
    if (!backupRes.ok && !force) {
      return { ok: false, error: 'backup failed: ' + backupRes.error, results: null };
    }
    var perCategory = [];
    var chain = Promise.resolve();
    for (var i = 0; i < SYNC_CATEGORIES.length; i++) {
      (function(cat) {
        if (!cat.legacyHeuristic || !cat.localKey) return;
        chain = chain.then(function() {
          var arr = _readArray(cat.localKey);
          var killIds = [];
          for (var k = 0; k < arr.length; k++) {
            if (cat.legacyHeuristic(arr[k])) killIds.push(arr[k][cat.idField]);
          }
          if (!killIds.length) {
            perCategory.push({ id: cat.id, count: 0, ok: true });
            return;
          }
          return tombstoneAndDeleteMany(cat.id, killIds).then(function(res) {
            perCategory.push({ id: cat.id, count: res.count, ok: res.ok, failed: res.failed });
          });
        });
      })(SYNC_CATEGORIES[i]);
    }
    return chain.then(function() {
      var allOk = perCategory.every(function(r) { return r.ok; });
      if (allOk) localStorage.setItem('roweos_focus_purge_v32', 'done');
      return { ok: allOk, results: perCategory, backup: backupRes };
    });
  });
}
window.purgeLegacyFocusResidue = purgeLegacyFocusResidue;

function restoreFromPrePurgeBackup_v32() {
  var raw = localStorage.getItem('roweos_pre_purge_backup_v32');
  if (!raw) {
    return Promise.resolve({ ok: false, error: 'no local backup; try cloud restore' });
  }
  try {
    var snap = JSON.parse(raw);
    var restored = 0;
    for (var k in snap) {
      if (!snap.hasOwnProperty(k)) continue;
      if (k.charAt(0) === '_') continue;
      try { localStorage.setItem(k, snap[k]); restored++; } catch (e) {}
    }
    localStorage.removeItem('roweos_focus_purge_v32');
    return Promise.resolve({ ok: true, restored: restored, capturedAt: snap._capturedAt });
  } catch (e2) {
    return Promise.resolve({ ok: false, error: e2 });
  }
}
window.restoreFromPrePurgeBackup_v32 = restoreFromPrePurgeBackup_v32;

// logos are stripped from brand.logo to '' for the localStorage write, with the
// full base64 stored in IndexedDB at roweos_brand_logo_<id>(_light) and pushed
// to a parallel Firestore subcollection brand_logos/<id>.

var LOGO_SIZE_THRESHOLD = 100 * 1024;

function _isOversize(dataUrl) {
  if (!dataUrl || typeof dataUrl !== 'string') return false;
  return dataUrl.length >= LOGO_SIZE_THRESHOLD;
}

function _idbLogoKey(brandId, isLight) {
  return 'roweos_brand_logo_' + brandId + (isLight ? '_light' : '');
}

function readBrandLogoSync(brand, opts) {
  if (!brand) return '';
  var isLight = opts && opts.light;
  var inline = isLight ? brand.logoLight : brand.logo;
  if (inline) return inline;
  if (!brand.id) return '';
  // Async fetch from IDB; caller may want to retry after a microtask
  var idbGet = window._idbGet;
  if (typeof idbGet === 'function') {
    var key = _idbLogoKey(brand.id, isLight);
    var maybe = idbGet(key);
    if (maybe && typeof maybe === 'string') return maybe; // sync cache
  }
  return '';
}
window.readBrandLogoSync = readBrandLogoSync;

function readBrandLogoAsync(brand, opts) {
  if (!brand || !brand.id) return Promise.resolve('');
  var isLight = opts && opts.light;
  var inline = isLight ? brand.logoLight : brand.logo;
  if (inline) return Promise.resolve(inline);
  var idbGet = window._idbGet;
  if (typeof idbGet !== 'function') return Promise.resolve('');
  return Promise.resolve(idbGet(_idbLogoKey(brand.id, isLight))).then(function(v) { return v || ''; });
}
window.readBrandLogoAsync = readBrandLogoAsync;

function writeBrandLogo(brand, dataUrl, opts) {
  if (!brand || !brand.id) return Promise.resolve({ ok: false, error: 'brand.id required' });
  var isLight = opts && opts.light;
  var idbPut = window._idbPut;
  if (_isOversize(dataUrl)) {
    if (isLight) { brand.logoLight = ''; brand.logoLightOversize = true; }
    else { brand.logo = ''; brand.logoOversize = true; }
    var p = idbPut ? Promise.resolve(idbPut(_idbLogoKey(brand.id, isLight), dataUrl)) : Promise.resolve();
    var writeDB = window.writeDB;
    if (writeDB) {
      var payload = isLight ? { logoLight: dataUrl, _modifiedAt: Date.now() } : { logo: dataUrl, _modifiedAt: Date.now() };
      p = p.then(function() { return writeDB('brand_logos/' + brand.id, payload); });
    }
    return p.then(function() { return { ok: true, oversize: true }; }, function(err) { return { ok: false, error: err }; });
  }
  // small — inline on the brand
  if (isLight) { brand.logoLight = dataUrl; brand.logoLightOversize = false; }
  else { brand.logo = dataUrl; brand.logoOversize = false; }
  // Clear any old IDB entry to avoid drift
  if (idbPut) { try { idbPut(_idbLogoKey(brand.id, isLight), null); } catch (e) {} }
  return Promise.resolve({ ok: true, oversize: false });
}
window.writeBrandLogo = writeBrandLogo;
```

function _handleOpenAIImageEdit(prompt, attachment, opts) {
  opts = opts || {};
  var apiKeys = (function() {
    try { return JSON.parse(localStorage.getItem('roweos_api_keys') || '{}'); }
    catch (e) { return {}; }
  })();
  var key = apiKeys.openai;
  if (!key) {
    showToast && showToast('OpenAI key required for GPT Image 2 edits', 'error');
    return Promise.resolve({ ok: false, error: 'no key' });
  }
  // Convert dataURL to Blob
  var dataUrl = attachment.dataUrl || attachment.url;
  if (!dataUrl) return Promise.resolve({ ok: false, error: 'no dataUrl on attachment' });
  var bin = atob(dataUrl.split(',')[1]);
  var arr = new Uint8Array(bin.length);
  for (var i = 0; i < bin.length; i++) arr[i] = bin.charCodeAt(i);
  var blob = new Blob([arr], { type: attachment.type || 'image/png' });
  var fd = new FormData();
  fd.append('image', blob, attachment.name || 'source.png');
  fd.append('prompt', prompt);
  fd.append('model', opts.model || 'gpt-image-2');
  return fetch('https://api.openai.com/v1/images/edits', {
    method: 'POST',
    headers: { 'Authorization': 'Bearer ' + key },
    body: fd
  }).then(function(r) { return r.json(); }).then(function(json) {
    if (json && json.data && json.data[0] && json.data[0].b64_json) {
      return { ok: true, dataUrl: 'data:image/png;base64,' + json.data[0].b64_json, provider: 'openai', model: opts.model };
    }
    return { ok: false, error: (json && json.error && json.error.message) || 'unknown error' };
  }, function(err) {
    return { ok: false, error: err };
  });
}
