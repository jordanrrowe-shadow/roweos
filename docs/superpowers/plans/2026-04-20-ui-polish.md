# UI Polish Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Two UI polish enhancements: (A) Studio inline results panel so AI Generate shows results in a collapsible panel below the button instead of just a toast, and (B) Pulse drag-and-drop goal reordering using native HTML5 drag events.

**Architecture:** Studio lives in `src/js/core/13-studio.js` with HTML in `src/html/brand/02-studio.html`. The `generateBrandAIRecommendations()` function (line 1160) currently generates operations and shows a toast. The results should also appear in a collapsible panel. Pulse goals are rendered via `renderPulseGoalsTab()` -> `renderPulse3Checklists()` in `src/js/core/25-documents-lifeai.js`. Goals are stored in `pulseGoals` array and saved via `savePulseGoals()`. Each goal card has class `pulse-3-checklist-card` with `data-goal-id`.

**Tech Stack:** Vanilla ES5 JavaScript, HTML5 Drag and Drop API, single-file HTML app (modular source in `src/`)

**Critical Constraints:**
- ES5 only (no arrow functions, let/const, template literals)
- Edit `src/` files only, then `bash src/build.sh` to regenerate dist
- No emojis in UI - SVG icons only
- No em-dashes in UI text
- No one-sided borders for accents/highlights

```javascript
// v29.3: Studio inline results panel

var _studioLastGeneratedOps = []; // v29.3: Last batch of generated operations

function showStudioResultsPanel(operations) { // v29.3:
  _studioLastGeneratedOps = operations || [];
  var panel = document.getElementById('studioResultsPanel');
  var body = document.getElementById('studioResultsBody');
  var countEl = document.getElementById('studioResultsCount');
  if (!panel || !body) return;

  countEl.textContent = operations.length;
  panel.classList.add('visible');
  panel.classList.remove('collapsed');

  var html = '';
  for (var i = 0; i < operations.length; i++) {
    var op = operations[i];
    html += '<div class="studio-result-card" onclick="toggleStudioResultCard(this)">';
    html += '<div class="studio-result-name">' + escapeHtml(op.name || 'Untitled') + '</div>';
    html += '<div class="studio-result-desc">' + escapeHtml(op.desc || '') + '</div>';
    html += '<span class="studio-result-category">' + escapeHtml(op.category || 'general') + '</span>';
    if (op.outputs && op.outputs.length > 0) {
      html += '<div class="studio-result-outputs"><ul>';
      for (var j = 0; j < op.outputs.length; j++) {
        html += '<li>' + escapeHtml(op.outputs[j]) + '</li>';
      }
      html += '</ul></div>';
    }
    html += '</div>';
  }
  body.innerHTML = html;
}

function toggleStudioResultsPanel() { // v29.3:
  var panel = document.getElementById('studioResultsPanel');
  if (!panel) return;
  panel.classList.toggle('collapsed');
}

function toggleStudioResultCard(cardEl) { // v29.3:
  if (!cardEl) return;
  cardEl.classList.toggle('expanded');
}

function dismissStudioResults() { // v29.3:
  var panel = document.getElementById('studioResultsPanel');
  if (panel) {
    panel.classList.remove('visible');
  }
  _studioLastGeneratedOps = [];
}
```

Find the `function togglePulseGoalLayout()` function (around line 8407). Insert AFTER it:

```javascript
// v29.3: Pulse goal drag-and-drop reordering
var _pulseDragGoalId = null;

function onPulseGoalDragStart(e) { // v29.3:
  var card = e.target.closest('.pulse-3-checklist-card');
  if (!card) return;
  _pulseDragGoalId = card.getAttribute('data-goal-id');
  card.classList.add('dragging');
  e.dataTransfer.effectAllowed = 'move';
  e.dataTransfer.setData('text/plain', _pulseDragGoalId);
  // Slight delay to allow the drag image to form before changing opacity
  setTimeout(function() {
    if (card) card.style.opacity = '0.4';
  }, 0);
}

function onPulseGoalDragOver(e) { // v29.3:
  e.preventDefault();
  e.dataTransfer.dropEffect = 'move';
  var card = e.target.closest('.pulse-3-checklist-card');
  if (!card) return;
  var dragId = card.getAttribute('data-goal-id');
  if (dragId === _pulseDragGoalId) return;
  // Remove drag-over from all, add to this one
  var allCards = document.querySelectorAll('.pulse-3-checklist-card.drag-over');
  for (var i = 0; i < allCards.length; i++) {
    allCards[i].classList.remove('drag-over');
  }
  card.classList.add('drag-over');
}

function onPulseGoalDragLeave(e) { // v29.3:
  var card = e.target.closest('.pulse-3-checklist-card');
  if (card) card.classList.remove('drag-over');
}

function onPulseGoalDrop(e) { // v29.3:
  e.preventDefault();
  var targetCard = e.target.closest('.pulse-3-checklist-card');
  if (!targetCard) return;
  var targetId = targetCard.getAttribute('data-goal-id');
  if (!_pulseDragGoalId || targetId === _pulseDragGoalId) return;

  // Find source and target indices in pulseGoals
  var sourceIdx = -1;
  var targetIdx = -1;
  for (var i = 0; i < pulseGoals.length; i++) {
    if (pulseGoals[i].id === _pulseDragGoalId) sourceIdx = i;
    if (pulseGoals[i].id === targetId) targetIdx = i;
  }
  if (sourceIdx === -1 || targetIdx === -1) return;

  // Remove source goal and insert at target position
  var movedGoal = pulseGoals.splice(sourceIdx, 1)[0];
  // Adjust target index if source was before target
  if (sourceIdx < targetIdx) targetIdx--;
  pulseGoals.splice(targetIdx, 0, movedGoal);

  // Update _modifiedAt for ordering persistence
  var now = Date.now();
  for (var j = 0; j < pulseGoals.length; j++) {
    pulseGoals[j]._sortOrder = j;
    pulseGoals[j]._modifiedAt = now;
  }

  savePulseGoals();

  // Clean up drag states
  _cleanupPulseGoalDragStates();
  _pulseDragGoalId = null;

  // Re-render
  renderPulseGoalsTab();
  if (typeof showToast === 'function') showToast('Goal order updated', 'success');
}

function onPulseGoalDragEnd(e) { // v29.3:
  _cleanupPulseGoalDragStates();
  _pulseDragGoalId = null;
}

function _cleanupPulseGoalDragStates() { // v29.3:
  var cards = document.querySelectorAll('.pulse-3-checklist-card');
  for (var i = 0; i < cards.length; i++) {
    cards[i].classList.remove('dragging', 'drag-over');
    cards[i].style.opacity = '';
  }
}
```
